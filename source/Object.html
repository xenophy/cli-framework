/**
 * @class CLI.Object
 * @singleton
 */

(function() {

    "use strict";

    var CLIObject = CLI.Object = {

        // {{{ each

        each: function(object, fn, scope) {

            var i, property;

            scope = scope || object;

            for (property in object) {

                if (object.hasOwnProperty(property)) {

                    if (fn.call(scope, property, object[property], object) === false) {

                        return;

                    }

                }

            }

        },

        // }}}

    };

})();
